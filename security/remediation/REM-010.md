## Recommended Fix

Implement a standard WASM ABI for input/output:

```rust
async fn call_entry_point(
    &self,
    store: &mut Store<HostState>,
    instance: &Instance,
    name: &str,
    input: &[u8],
) -> Result<Vec<u8>> {
    let memory = instance.get_memory(store, "memory")
        .ok_or_else(|| GhostError::wasm("Memory export not found"))?;

    // Get the WASM module's allocator
    let alloc = instance.get_typed_func::<i32, i32>(store, "alloc")
        .map_err(|_| GhostError::wasm("Module must export 'alloc(size: i32) -> ptr: i32'"))?;

    // Allocate memory for input
    let input_ptr = alloc.call_async(store, input.len() as i32).await
        .map_err(|e| GhostError::wasm(format!("alloc failed: {}", e)))?;

    // Write input data
    memory.write(store, input_ptr as usize, input)?;

    // Call the entry point with (input_ptr, input_len)
    let func = instance.get_typed_func::<(i32, i32), i32>(store, name)?;
    let result_ptr = func.call_async(store, (input_ptr, input.len() as i32)).await?;

    // Read result length from the first 4 bytes at result_ptr
    let mut len_bytes = [0u8; 4];
    memory.read(store, result_ptr as usize, &mut len_bytes)?;
    let result_len = u32::from_le_bytes(len_bytes) as usize;

    // Validate result length
    if result_len > 1_048_576 {
        return Err(GhostError::wasm("Result too large (> 1MB)"));
    }

    // Read result data
    let mut output = vec![0u8; result_len];
    memory.read(store, (result_ptr as usize) + 4, &mut output)?;

    Ok(output)
}
```
