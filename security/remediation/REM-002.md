struct NonceTracker {
    seen: HashSet<u64>,
    expiry: HashMap<u64, Instant>,
}

impl NonceTracker {
    fn check_and_record(&mut self, nonce: u64) -> bool {
        self.cleanup_expired();
        if self.seen.contains(&nonce) {
            return false; // Replay detected
        }
        self.seen.insert(nonce);
        self.expiry.insert(nonce, Instant::now() + Duration::from_secs(300));
        true
    }
}
