```rust
use url::Url;

pub async fn fetch_url(&self, url: &str, whitelist: &[String]) -> Result<Vec<u8>, NetworkError> {
    if !self.has_capability("network") {
        return Err(NetworkError::CapabilityDenied);
    }

    // Parse and validate URL
    let parsed = Url::parse(url).map_err(|_| NetworkError::NotWhitelisted)?;

    // Only allow HTTPS
    if parsed.scheme() != "https" {
        return Err(NetworkError::NotWhitelisted);
    }

    // Block internal/private IPs
    if is_private_ip(parsed.host_str().unwrap_or("")) {
        return Err(NetworkError::NotWhitelisted);
    }

    // Check against whitelist patterns
    let allowed = whitelist.iter().any(|pattern| {
        matches_url_pattern(pattern, url)
    });

    if !allowed {
        return Err(NetworkError::NotWhitelisted);
    }

    // Perform the fetch with timeout
    let client = reqwest::Client::builder()
        .timeout(std::time::Duration::from_secs(10))
        .redirect(reqwest::redirect::Policy::limited(3))
        .build()
        .map_err(|e| NetworkError::RequestFailed(e.to_string()))?;

    let response = client.get(url).send().await
        .map_err(|e| NetworkError::RequestFailed(e.to_string()))?;

    let bytes = response.bytes().await
        .map_err(|e| NetworkError::RequestFailed(e.to_string()))?;

    // Limit response size (1MB max)
    if bytes.len() > 1_048_576 {
        return Err(NetworkError::RequestFailed("Response too large".into()));
    }

    Ok(bytes.to_vec())
}

fn is_private_ip(host: &str) -> bool {
    // Block localhost, 10.x, 172.16-31.x, 192.168.x, 169.254.x
    host.starts_with("localhost")
        || host.starts_with("127.")
        || host.starts_with("10.")
        || host.starts_with("192.168.")
        || host == "0.0.0.0"
        || host == "[::]"
        || host == "[::1]"
}
```
